  <script>
    async function generateIdea() {
      const domain = document.getElementById("domain").value;
      const audience = document.getElementById("audience").value;

      try {
        // Send a POST request to the Vercel API
        const response = await fetch('/api/generateIdea', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({ domain, audience })
        });

        const data = await response.json();
        
        // Display the generated idea
        const generatedText = data.idea.split('\n');
        document.getElementById("pitch").innerText = generatedText[0] || 'No pitch available';
        document.getElementById("mvp").innerText = generatedText[1] || 'No MVP available';
        document.getElementById("money").innerText = generatedText[2] || 'No monetization available';

        document.getElementById("output").style.display = "block";
      } catch (error) {
        console.error('Error:', error);
        alert('Something went wrong! Please try again.');
      }
    }
  </script>
